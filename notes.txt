public class CommonUtils {
    // Static final map initialized once with all status mappings
    private static final Map<String, String> STATUS_MAP = new HashMap<>() {{
        put("0", "ORDER ENTRY");
        put("1", "INIT. SET");
        put("3", "CORR. SET");
        put("4", "QUEUE REA");
        put("5", "RECV IN-P");
        put("6", "RECV ENDE");
        put("7", "PAC REJEC");
        put("9", "RESEND RE");
        put("10", "ORDER COM");
        put("12", "PAC RECPT");
        put("14", "PAC ACCEP");
        put("98", "REPL COM");
        put("99", "ORDER COM");
    }};

    private static final DateTimeFormatter DATE_FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM-dd");

    public static String getStatusInMedqa(String statusCode) { // usage ยน Aravinda
        if (statusCode == null || statusCode.trim().isEmpty()) {
            log.info("Invalid input: null or empty status code");
            return null;
        }
        return STATUS_MAP.get(statusCode);
    }

    // Other methods like isBackdateDateValid can remain as previously optimized

    private static final Logger log = LoggerFactory.getLogger(CommonUtils.class);
}
