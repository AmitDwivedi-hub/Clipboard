public class CommonUtils {
    private static final DateTimeFormatter DATE_FORMATTER = DateTimeFormatter.ofPattern("yyyy-MM-dd");

    // Constants for substring indices and delimiter
    private static final int UUID_START_INDEX = 1;
    private static final int UUID_MIDDLE_INDEX = 8;
    private static final int UUID_END_INDEX = 18;
    private static final String DELIMITER = "-";

    public static String getTicketId(String orderType) { // ยน Aravinda Perumbudi *
        if (orderType == null || orderType.trim().isEmpty()) {
            log.info("Invalid input: null or empty orderType");
            return null;
        }

        // Generate UUID and convert to string, removing hyphens
        String uuidStr = UUID.randomUUID().toString().replace(DELIMITER, "");

        // Extract relevant parts of UUID and concatenate with orderType
        return orderType.toUpperCase() + DELIMITER +
               uuidStr.substring(UUID_START_INDEX, UUID_MIDDLE_INDEX) + DELIMITER +
               uuidStr.substring(UUID_MIDDLE_INDEX, UUID_END_INDEX) + DELIMITER +
               uuidStr.substring(UUID_END_INDEX);
    }

    // Other methods like isBackdateDateValid can remain as previously optimized

    private static final Logger log = LoggerFactory.getLogger(CommonUtils.class);
}
